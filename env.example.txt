# Скопируйте этот файл в .env и отредактируйте значения:  cp env.example.txt .env
#
# OpenAI-совместимый endpoint вашего Qwen3 (БЕЗ /chat/completions — SDK добавит сам).
# Ollama:    http://localhost:11434/v1
# vLLM:      http://localhost:8000/v1
# LM Studio: http://localhost:1234/v1
QWEN_BASE_URL=http://localhost:11434/v1

# Имя модели, как она зарегистрирована в раннере.
# Ollama:  qwen3:8b  / qwen3:14b  / qwen3:32b
# vLLM:    Qwen/Qwen3-8B  (или путь/имя, под которым запущена модель)
QWEN_MODEL=qwen3:8b

# Ключ. Для локальных раннеров обычно подходит любая непустая строка.
# Для некоторых self-hosted endpoint'ов ключ не требуется — оставьте как есть или пусто.
QWEN_API_KEY=local

# Максимальный объём контекста модели в токенах (окно, с которым реально запущен раннер).
# ВАЖНО: в Ollama по умолчанию 4096 — поднимите через параметр num_ctx при запуске
# модели, иначе длинные документы будут резаться. Здесь укажите фактическое окно.
MAX_CONTEXT=32000

# Доля окна, резервируемая под промпт-инструкции и ответ (0..1).
# Под документы остаётся MAX_CONTEXT * (1 - RESERVE_RATIO).
RESERVE_RATIO=0.35

# Температура генерации (0 — максимально детерминированно).
TEMPERATURE=0.2

# Таймаут запроса к модели, сек.
REQUEST_TIMEOUT=600

# Потолок числа пар секций в режиме посекционного сравнения (защита от DoS:
# битый/огромный docx не должен порождать тысячи запросов к LLM).
MAX_SECTION_PAIRS=300

# Mock-режим: если true, реальные запросы к LLM не выполняются — возвращается
# заглушка. Удобно для проверки UI и парсинга документов без запущенной модели.
# Уберите или оставьте пустым для работы с настоящим Qwen3.
MOCK_LLM=

# --- Прокси для УСТАНОВКИ зависимостей (pip) ---------------------------------
# Если на рабочем компе выход в интернет только через корпоративный прокси,
# укажите его здесь — run.bat подхватит это значение ДО запуска pip install.
# Оставьте пустым, если прокси не нужен.
# Формат:  http://[ПОЛЬЗОВАТЕЛЬ:ПАРОЛЬ@]адрес:порт
# Пример:  http://proxy.company.local:8080
#          http://ivanov:secret@proxy.company.local:8080
PIP_PROXY=

# Если прокси подменяет TLS-сертификаты (self-signed) и pip падает с ошибкой
# SSL/CERTIFICATE_VERIFY_FAILED — поставьте 1, чтобы доверять зеркалам pip.
# Менее безопасно; включайте только если без этого установка не идёт.
PIP_TRUST_HOST=0
